// JavaScript pour gérer le panier et mettre à jour l’interface utilisateur
let cart = {};

// Function to add fruit to basket
function addToCart(fruit) {
    if (!cart[fruit]) {
        cart[fruit] = 1;
    } else {
        cart[fruit]++;
    }
    return cart[fruit];
}
